**创建表语句**

```sql
CREATE TABLE Pet (
	id INT PRIMARY KEY AUTO_INCREMENT,
	`name` VARCHAR ( 50 ) NOT NULL,
	species VARCHAR ( 50 ) NOT NULL,
	age INT NOT NULL,
	adopt_status BOOLEAN DEFAULT FALSE,
	added_date DATETIME DEFAULT CURRENT_TIMESTAMP 
);

CREATE TABLE Adopter (
	id INT PRIMARY KEY AUTO_INCREMENT,
	`name` VARCHAR ( 50 ) NOT NULL,
	phone VARCHAR ( 20 ) NOT NULL,
	address VARCHAR ( 100 ) NOT NULL,
	registration_date DATETIME DEFAULT CURRENT_TIMESTAMP,
	`password` VARCHAR ( 50 ) NOT NULL 
);

CREATE TABLE Adoption_Record (
	id INT PRIMARY KEY AUTO_INCREMENT,
	pet_id INT,
	adopter_id INT,
	adoption_date DATETIME DEFAULT CURRENT_TIMESTAMP,
	`status` VARCHAR ( 20 ) ,
	CHECK (status IN ('申请中', '已批准', '已完成')),
	FOREIGN KEY ( pet_id ) REFERENCES Pet ( id ) ON DELETE RESTRICT ON UPDATE RESTRICT,
FOREIGN KEY ( adopter_id ) REFERENCES Adopter ( id ) ON DELETE RESTRICT ON UPDATE RESTRICT 
);
```

==思考==
建表时某些字段的约束并不了解，比如自动设置当前时间、强制字段名选择等，另外需要注意自动设置当前时间也可以在java代码实现管理系统数据访问层时添加该功能，不过就在表的约束上添加更为直观、不易遗忘
