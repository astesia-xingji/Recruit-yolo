**触发器**
```sql
-- 当删除宠物记录时，自动删除相关的领养契约记录
DELIMITER $$
CREATE TRIGGER delete_pet_adoption_records
BEFORE DELETE ON Pet 
FOR EACH ROW
BEGIN
    DELETE FROM Adoption_Record 
    WHERE pet_id = OLD.id; 
END$$
DELIMITER ;

-- 当领养人注销（删除记录）时，自动删除其所有领养契约记录
DELIMITER $$
CREATE TRIGGER delete_adopter_adoption_records
BEFORE DELETE ON Adopter 
FOR EACH ROW
BEGIN
    DELETE FROM Adoption_Record 
    WHERE adopter_id = OLD.id; 
END$$
DELIMITER ;
```
